% Y Combinator  
(defun Y (f)
 ((位 (g) (funcall g g))
   (位 (x)
     (funcall f (位 (&rest a)
                  (apply (funcall x x) a))))))

% fibonacci    
(defun fib (n)
  (funcall
   (Y (lambda (f)
       (lambda (n a b)
         (if (< n 1)
           a
           (funcall f (1- n) b (+ a b))))))
   n 0 1))                  

% factorial
(defun fac (n)
  (funcall
   (Y (lambda (f)
       (lambda (n)
         (if (zerop n)
	   1
	   (* n (funcall f (1- n)))))))
   n))
\end{comment}
