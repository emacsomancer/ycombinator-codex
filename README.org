#+begin_src lisp
  ;; Y Combinator  
  (defun Y (f)
   ((lambda (g) (funcall g g))
     (lambda (x)
       (funcall f (lambda (&rest a)
                    (apply (funcall x x) a))))))

  ;; fibonacci    
  (defun fib (n)
    (funcall
     (Y (lambda (f)
         (lambda (n a b)
           (if (< n 1)
             a
             (funcall f (1- n) b (+ a b))))))
     n 0 1))

  ;; factorial
  (defun fac (n)
    (funcall
     (Y (lambda (f)
         (lambda (n)
           (if (zerop n)
             1
             (* n (funcall f (1- n)))))))
     n))

? (mapcar #'fib '(1 2 3 4 5 6 7 8 9))
(1 1 2 3 5 8 13 21 34)

? (mapcar #'fac '(1 2 3 4 5 6 7 8 9))
(1 2 6 24 120 720 5040 40320 362880)        
#+end_src
